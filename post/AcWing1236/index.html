    {
        int l = 0 , r = n - 1 , mid;
        long long x , y;
        while(l < r)
        {
            mid = l + r + 1 >> 1;
            if(a[mid] < b[i]) l = mid;
            else r = mid - 1;
        }
        if(a[l] < b[i]) x = l + 1;
        else x = 0;

        while(l < r)
        {
            mid = l + r >> 1;
            if(c[mid] > b[i]) r = mid;
            else l = mid + 1;
        }

        res += x * y;
    }
    
    printf("%lld" , res);
    
    return 0;
}
