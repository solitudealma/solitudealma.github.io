#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 1e5 + 10;

int n , k;
int a[N];

int quick_sort(int q[] , int l , int r , int k)
{
    if(l >= r) return q[l];
    
    int i = l - 1 , j = r + 1 , x = q[l + r >> 1];
    while(i < j)
    {
        do ++ i; while(q[i] < x);
        do -- j; while(q[j] > x);
        if(i < j) swap(q[i] , q[j]);
    }
    
    int len = j - l + 1;
    if(len >= k) return quick_sort(q , l , j , k);
    else return quick_sort(q , j + 1 , r , k - len);
}

int main()
{
    
    scanf("%d%d" , &n , &k);
    
    for(int i = 0; i < n; ++ i) scanf("%d" , &a[i]);
    
    int res = quick_sort(a , 0 , n - 1 , k);
    
    printf("%d" , res);
    
    return 0;
}
